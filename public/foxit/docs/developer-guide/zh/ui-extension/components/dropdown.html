# 分离式

这里的分离是指 下拉按钮上面的图标文字和下拉箭头分离; 分离式的下拉按钮，只有点击下拉箭头才会显示下拉列表，通常被用来当做select组件来使用，默认Dropdown组件是分离式的，具体情况请参考下面的例子来对比差异：

<html>
    <template id="layout-template">
        <webpdf>
            <div>
                <!-- By default, the value of dropdown's separate option is true -->
                <dropdown name="separate-dropdown" icon-class="fv__icon-toolbar-square" text="Separated Dropdown" selected="0">
                    <xbutton name="separate-dropdown-square-btn" icon-class="fv__icon-toolbar-square">Square</xbutton>
                    <xbutton icon-class="fv__icon-toolbar-circle">Circle</xbutton>
                    <file-selector>Select A File</file-selector>
                    <li class="my-dropdown-list-item">
                        html &lt;li&gt; tag
                    </li>
                </dropdown>
                <dropdown name="non-separate-dropdown" icon-class="fv__icon-toolbar-shape" text="not separated Dropdown" separate="false">
                    <xbutton icon-class="fv__icon-toolbar-square">Square</xbutton>
                    <xbutton icon-class="fv__icon-toolbar-circle">Circle</xbutton>
                </dropdown>
            </div>
            <div class="fv__ui-body">
                <viewer></viewer>
            </div>
        </webpdf>
    </template>
</html>
<style>
    .my-dropdown-list-item {
        padding: 10px 0;
        text-align: center;
    }
    .fv__ui-dropdown {
        width: auto;
    }
</style>
<script>
    var CustomAppearance = UIExtension.appearances.Appearance.extend({
        getLayoutTemplate: function() {
            return document.getElementById('layout-template').innerHTML;
        },
        getDefaultFragments: function() {
            return [{
                target: 'separate-dropdown-square-btn',
                config: [{
                    callback: function() {
                        // dropdown 配置 selected="0", 这样直接点击下拉按钮图标文字就可以触发被选中的下拉列表项的事件。
                        alert('Click on Separate Dropdown');
                    }
                }]
            }];
        },
        disableAll: function(){}
    });
    var libPath = window.top.location.origin + '/lib';
    var pdfui = new UIExtension.PDFUI({
            viewerOptions: {
                libPath: libPath,
                jr: {
                    licenseSN: licenseSN,
                    licenseKey: licenseKey
                }
            },
            renderTo: document.body,
            appearance: CustomAppearance,
            addons: []
    });
</script>

{
    "iframeOptions": {
        "style": "height: 500px"
    }

# 可编辑

<html>
    <template id="layout-template">
        <webpdf>
            <div>
                <dropdown name="font-editable-dropdown" editable>
                    <xbutton>Helvetica</xbutton>
                    <xbutton>Courier</xbutton>
                    <xbutton>Times-Bold</xbutton>
                    <xbutton>宋体</xbutton>
                </dropdown>
                <dropdown name="zoom-editable-dropdown" editable @controller="custom:ZoomPageController">
                    <xbutton @controller="custom:ScaleRatioController" scale="0.5">50%</xbutton>
                    <xbutton @controller="custom:ScaleRatioController" scale="0.75">75%</xbutton>
                    <xbutton @controller="custom:ScaleRatioController" scale="1">100%</xbutton>
                </dropdown>
            </div>
            <div class="fv__ui-body">
                <viewer></viewer>
            </div>
        </webpdf>
    </template>
</html>
<style>
    .fv__ui-dropdown {
        width: 80px;
    }
</style>
<script>
    UIExtension.PDFUI.module('custom', [])
    .controller('ScaleRatioController', {
        handle: function() {
            const scaleRatio = parseFloat(this.component.element.getAttribute('scale'));
            this.component.parent.setEditValue(scaleRatio);
        }
    })
    .controller('ZoomPageController', {
        mounted: function() {
            const component = this.component;
            const firstChild = component.childAt(0);
            const scaleRatio = parseFloat(firstChild.element.getAttribute('scale'))
            component.setEditValue(scaleRatio);
            component.on('change', function(newValue, oldValue) {
                if(isNaN(parseFloat(newValue))) {
                    alert('Illegal scale value: ' + newValue);
                    component.setEditValue(oldValue);
                    return;
                }
                alert('scale value changed to: ' + newValue)
            })
        }
    });
    var CustomAppearance = UIExtension.appearances.Appearance.extend({
        getLayoutTemplate: function() {
            return document.getElementById('layout-template').innerHTML;
        },
        getDefaultFragments() {
            return [{
                target: 'zoom-editable-dropdown',
                config: {
                    editOptions: {
                        type: 'number',
                        min: 0,
                        max: 10,
                        step: 0.1
                    }
                }
            }];
        },
        disableAll: function(){}
    });
    var libPath = window.top.location.origin + '/lib';
    var pdfui = new UIExtension.PDFUI({
            viewerOptions: {
                libPath: libPath,
                jr: {
                    licenseSN: licenseSN,
                    licenseKey: licenseKey
                }
            },
            renderTo: document.body,
            appearance: CustomAppearance,
            addons: []
    });
</script>

{
    "iframeOptions": {
        "style": "height: 500px"
    }
}

# 动态列表

<html>
    <template id="layout-template">
        <webpdf>
            <div>
                <dropdown separate="false" @controller="custom:DropdownItemListController as ctrl" text="Dynamic dropdown list">
                    <li style="padding-left: 1em;">Click button to create more</li>
                    <li @foreach="item in ctrl.items track by id">
                        <text @sync.text="item.text"></text>
                    </li>
                    <xbutton icon-class="fv__icon-toolbar-add-sign" @controller="custom:AddItemController">Add dropdown item</xbutton>
                </dropdown>
            </div>
            <div class="fv__ui-body">
                <viewer></viewer>
            </div>
        </webpdf>
    </template>
</html>
<style>
    .fv__ui-dropdown {
        width: auto;
    }
</style>
<script>
    UIExtension.PDFUI.module('custom', [])
        .controller('DropdownItemListController', {
            init: function() {
                this.items = [{
                    id: Date.now().toString(16),
                    text: new Date().toLocaleString()
                }];
            },
            addItem: function(data) {
                this.items = this.items.concat(data);
                this.digest();
            }
        })
        .controller('AddItemController', {
            handle: function() {
                const itemListCtrl = this.data.ctrl;
                itemListCtrl.addItem({
                    id: Date.now().toString(16),
                    text: new Date().toLocaleString()
                });
            }
        })
    var CustomAppearance = UIExtension.appearances.Appearance.extend({
        getLayoutTemplate() {
            return document.getElementById('layout-template').innerHTML;
        },
        disableAll(){}
    });
    var libPath = window.top.location.origin + '/lib';
    var pdfui = new UIExtension.PDFUI({
            viewerOptions: {
                libPath: libPath,
                jr: {
                    licenseSN: licenseSN,
                    licenseKey: licenseKey
                }
            },
            renderTo: document.body,
            appearance: CustomAppearance,
            addons: []
    });
</script>

{
    "iframeOptions": {
        "style": "height: 500px"
    }

# API

# 模板属性

组件模板示例

<dropdown text="" icon-class="" editable align="left" valign="bottom" separate="true" selected="0"></dropdown>
属性 说明 类型 默认值 版本
text 设置 dropdown 下拉按钮文本 string '' 7.0
icon-class 指定图标的css class string '' 7.0
editable 是否可编辑 boolean false 7.0
align 水平对齐 'left'|'right'|'out-right'|'out-left'|'center'|'client-center' 'left' 7.0
valign 垂直对齐 'top'|'bottom'|'center'|'client-center' 'bottom' 7.0
separate 下拉按钮和文字是否图标分离 boolean true 7.0

# fragment特殊配置属性

除了editOptions,其它属性和“模板属性”一致！

{
    target: 'dropdown-name',
    config: {
        editOptions: {
            type: 'text',
            min: 0,
            max: 0,
            step: 0,
            vallue: ''
        }
    }
}
属性 说明 类型 默认值 版本
editOptions.type 设置 dropdown 编辑框模式，支持 'text' 文本编辑 和 'number' 数字编辑 string 'text' 7.0
editOptions.min 编辑框最小数值，数字编辑模式是'number'时有效 number 7.0
editOptions.max 编辑框最大数值，数字编辑模式是'number'时有效 number 7.0
editOptions.step 编辑框数值的步长，数字编辑模式是'number'时有效 number 7.0
editOptions.value 编辑框初始值 string|number 7.0

对象属性

属性 说明 类型
disabled 按钮失效状态 boolean
isVisible 按钮显示状态 boolean
isActive 下拉列表是否开启 boolean

# 方法

方法定义 说明 版本
setEditValue(text: String|number): void 设置输入框值（不会触发change事件！） 7.0
disable(): void 禁用 dropdown，禁用后的 dropdown 组件将不可用 7.0
enable(): void 启用被禁用后的dropdown 7.0
show(): void 显示被隐藏的 dropdown 7.0
hide(): void 隐藏 dropdown 7.0
active(): void 显示下拉列表 7.0
deactive(): void 关闭下拉列表 7.0
destroy(): void 销毁组件 7.0

# Events

事件名称 说明 示例 版本
active 下拉列表展开时触发 dropdown.on('active', () => {}) 7.0
deactive 下拉列表隐藏时触发 dropdown.on('deactive', () => {}) 7.0
change 用户在输入框输入并失去焦点后触发 dropdown.on('change', (newValue,oldValue) => {}) 7.0

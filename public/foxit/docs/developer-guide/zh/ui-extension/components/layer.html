# Layer Component

Layer 通常用来容纳一些内容，并浮动在其它元素之上的组件，所以可以被用来实现 对话框、工具提示、右键菜单等等。

# 代码演示

# getting started

<html>
    <template id="layout-template">
        <webpdf>
            <div class="flex-container">
                <xbutton action="show-layer" @controller="custom:ShowHideLayerController">Click to show layer</xbutton>
                <xbutton action="hide-layer" @controller="custom:ShowHideLayerController">Click to hide layer</xbutton>
            </div>
            <div class="fv__ui-body">
                <viewer></viewer>
            </div>
            <template>
                <layer name="my-layer" class="center">
                    <text>Hello! I'm a layer component!</text>
                </layer>
            </template>
        </webpdf>
    </template>
</html>
<style>
    .flex-container {
        display: flex;
        justify-content: space-between;
    }
</style>
<script>
    UIExtension.PDFUI.module('custom', [])
        .controller('ShowHideLayerController', {
            handle: function() {
                const layer = this.getComponentByName('my-layer');
                const action = this.component.getAttribute('action');
                switch(action) {
                    case 'show-layer':
                        layer.show();
                        break;
                    case 'hide-layer':
                        layer.hide();
                        break;
                }
            }
        });
    var CustomAppearance = UIExtension.appearances.Appearance.extend({
        getLayoutTemplate: function() {
            return document.getElementById('layout-template').innerHTML;
        },
        disableAll: function(){}
    });
    var libPath = window.top.location.origin + '/lib';
    var pdfui = new UIExtension.PDFUI({
            viewerOptions: {
                libPath: libPath,
                jr: {
                    licenseSN: licenseSN,
                    licenseKey: licenseKey
                }
            },
            renderTo: document.body,
            appearance: CustomAppearance,
            addons: []
    });
</script>

            <div class="flex-container">
                <xbutton target-layer="my-layer" @controller="custom:ShowLayerController">Click to show layer</xbutton>

# 指定 layer 组件父节点

默认情况下， layer 组件显示时，会将DOM节点插入到根组件末尾， 这样有些情况下，layer 的 DOM 层级可能会不正确，可以指定 append-to 来改变插入节点的位置，请参考下面的示例：

<html>
    <template id="layout-template">
        <webpdf>
            <div class="flex-container">
                <xbutton action="show-layer" @controller="custom:ShowHideLayerController">Click to show layer</xbutton>
                <xbutton action="hide-layer" @controller="custom:ShowHideLayerController">Click to hide layer</xbutton>
            </div>
            <div class="fv__ui-body">
                <viewer></viewer>
            </div>
            <template>
                <layer name="my-layer" class="center">
                    <text>Hello! I'm a layer component!</text>
                </layer>
            </template>
        </webpdf>
    </template>
</html>
<style>
    .flex-container {
        display: flex;
        justify-content: space-between;
    }
</style>
<script>
    UIExtension.PDFUI.module('custom', [])
        .controller('ShowHideLayerController', {
            handle: function() {
                const layer = this.getComponentByName('my-layer');
                const action = this.component.getAttribute('action');
                switch(action) {
                    case 'show-layer':
                        layer.show(document.body); // 显示时指定插入到 document.body 上
                        break;
                    case 'hide-layer':
                        layer.hide();
                        break;
                }
            }
        });
    var CustomAppearance = UIExtension.appearances.Appearance.extend({
        getLayoutTemplate: function() {
            return document.getElementById('layout-template').innerHTML;
        },
        disableAll: function(){}
    });
    var libPath = window.top.location.origin + '/lib';
    var pdfui = new UIExtension.PDFUI({
            viewerOptions: {
                libPath: libPath,
                jr: {
                    licenseSN: licenseSN,
                    licenseKey: licenseKey
                }
            },
            renderTo: document.body,
            appearance: CustomAppearance,
            addons: []
    });
</script>

# API

# 组件模板

<layer class="center" visible modal backdrop>
    <layer-header title="" icon-class="fv__icon-toolbar-print"></layer-header>
</layer>

&ltlayer> 组件模板属性说明：

属性 说明 类型 默认值 版本
visible 是否立即显示 boolean false 7.0.0
modal 是否模态框 boolean false 7.0.0
backdrop 模态框是否使用黑色半透明背景 boolean false 7.0.0
class="center" 添加 center 后，layer 将居中显示 -- -- 7.0.0
class="centerv" 添加 centerv 后，layer 将垂直居中显示 -- -- 7.0.0
class="centerh" 添加 centerh 后，layer 将水平居中显示 -- -- 7.0.0
class="left" 添加 left 后，layer 将靠左显示 -- -- 7.0.0
class="right" 添加 right 后，layer 将靠右显示 -- -- 7.0.0
class="top" 添加 top 后，layer 将靠顶部显示 -- -- 7.0.0
class="bottom" 添加 bottom 后，layer 将靠底部显示 -- -- 7.0.0

<layer-header> 组件模板属性说明

属性 说明 类型 默认值 版本
title 标题内容 string '' 7.0.0
icon-class 标题图标 string '' 7.0.0

# 方法

<layer> 组件方法

方法定义 说明 版本
show(appendTo: HTMLElement): void 将 layer 组件插入到指定的DOM节点并显示 7.0.0
open(appendTo: HTMLElement): void 功能和show方法一样 7.0.0
hide(): void 隐藏layer组件 7.0.0
close(): void 隐藏layer组件并销毁 7.0.0

# Events

<layer> 事件

事件名称 说明 示例 版本
shown layer 显示后触发 layer.on('shown', () => void) 7.0.0
hidden layer 隐藏后触发 layer.on('hidden', () => void) 7.0.0
closed layer 隐藏并销毁（调用close()方法）后触发 layer.on('closed', () => void) 7.0.0
